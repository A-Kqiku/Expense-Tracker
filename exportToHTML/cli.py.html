from methods import *
from manager import *


app_status: bool= True
manager = Manager()

menu_text: str = "\n(1) Show Expenses \n(2) Add Expenses \n(3) Edit Expenses \n(4) Delete Expenses \n(5) Exit \n"


while app_status:
    user_input: int = int(input(menu_text))

    match user_input:
        case 1:
            show_expenses(manager)
        case 2:
            add_expenses(manager)
        case 3:
            show_expenses(manager)
            edit_expenses(manager)
        case 4:
            delete_expenses(manager)
        case 5:
            print("Bye")
            app_status = False
        case _:
            print("Invalid input")
